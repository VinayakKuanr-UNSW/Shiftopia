2. Final Tables
2.1 availability_rules (source of truth)
This stores what the user configured, not what the scheduler consumes.
create table availability_rules (
  id uuid primary key default gen_random_uuid(),

  profile_id uuid not null,

  start_date date not null,
  start_time time not null,
  end_time time not null,

  repeat_type text not null default 'none',
  -- 'none' | 'daily' | 'weekly' | 'fortnightly'

  repeat_days smallint[] null,
  -- 1=Mon ... 7=Sun
  -- required for weekly and fortnightly

  repeat_end_date date null,

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

Constraints (important)
alter table availability_rules
add constraint chk_time_order
check (start_time < end_time);

alter table availability_rules
add constraint chk_repeat_days_required
check (
  (repeat_type in ('weekly','fortnightly') and repeat_days is not null)
  or (repeat_type not in ('weekly','fortnightly'))
);

alter table availability_rules
add constraint chk_repeat_end_date
check (
  repeat_type = 'none'
  or repeat_end_date is not null
);


2.2 Indexes for availability_rules
These support:
list view
edit/delete
bulk cleanup
create index idx_availability_rules_profile
on availability_rules(profile_id);

create index idx_availability_rules_profile_created
on availability_rules(profile_id, created_at desc);


3. availability_slots (materialized, scheduler-ready)
This table is derived and replaceable.
create table availability_slots (
  id uuid primary key default gen_random_uuid(),

  rule_id uuid not null
    references availability_rules(id)
    on delete cascade,

  profile_id uuid not null,

  slot_date date not null,
  start_time time not null,
  end_time time not null,

  created_at timestamptz not null default now()
);


4. Critical Indexes for availability_slots
These are non-negotiable.
4.1 Calendar rendering (by user + month)
create index idx_slots_profile_date
on availability_slots(profile_id, slot_date);

Used by:
monthly calendar
daily view
mobile fetch

4.2 Scheduler lookup (time containment)
create index idx_slots_scheduler
on availability_slots(profile_id, slot_date, start_time, end_time);

Used by:
exists (
  select 1
  from availability_slots
  where profile_id = ?
    and slot_date = ?
    and start_time <= shift_start
    and end_time >= shift_end
)


4.3 Rule cleanup and regeneration
create index idx_slots_rule
on availability_slots(rule_id);

Used when:
editing availability
deleting availability
regenerating slots

5. Optional but recommended: uniqueness guard
This prevents duplicate slots during retries or race conditions.
create unique index ux_slots_dedup
on availability_slots (
  rule_id,
  slot_date,
  start_time,
  end_time
);

This makes slot generation idempotent.

6. Data lifecycle rules (very important)
Rule creation
Insert into availability_rules
Generate slots async
Insert into availability_slots
Rule edit
Delete rule (cascade deletes slots)
Insert new rule
Generate fresh slots
Rule delete
Delete rule
Slots auto removed
No partial updates. No slot patching.

7. Scheduler contract (locked)
Scheduler must only read availability_slots.
It must never inspect rules.
This keeps scheduling fast and predictable.
